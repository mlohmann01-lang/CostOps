services/trust_score.py
def calculate_trust_score(identity_confidence, freshness_score, signal_confidence):    score = (        identity_confidence * 0.4        + freshness_score * 0.3        + signal_confidence * 0.3    )    return round(score, 2)def execution_status_from_score(score):    if score >= 0.90:        return "APPROVAL_REQUIRED"    if score >= 0.75:        return "APPROVAL_REQUIRED"    if score >= 0.50:        return "INVESTIGATE"    return "BLOCKED"def score_user_reclaim_candidate(user):    email = user.get("userPrincipalName")    disabled = user.get("accountEnabled") is False    last_login_days = user.get("lastLoginDaysAgo", 999)    identity_confidence = 1.0 if email else 0.3    if last_login_days <= 7:        freshness_score = 1.0    elif last_login_days <= 30:        freshness_score = 0.7    else:        freshness_score = 0.4    if disabled and last_login_days > 90:        signal_confidence = 1.0    elif last_login_days > 90:        signal_confidence = 0.7    elif disabled:        signal_confidence = 0.7    else:        signal_confidence = 0.2    trust_score = calculate_trust_score(        identity_confidence,        freshness_score,        signal_confidence,    )    return {        "trust_score": trust_score,        "execution_status": execution_status_from_score(trust_score),        "identity_confidence": identity_confidence,        "freshness_score": freshness_score,        "signal_confidence": signal_confidence,    }

services/playbook_inactive_users.py
from services.trust_score import score_user_reclaim_candidateDEFAULT_LICENSE_COSTS = {    "E5": 57.00,    "E3": 36.00,    "BUSINESS_PREMIUM": 32.00,}def is_excluded_user(user):    email = user.get("userPrincipalName", "").lower()    display_name = user.get("displayName", "").lower()    exclusion_terms = [        "admin",        "service",        "shared",        "noreply",        "no-reply",    ]    return any(term in email or term in display_name for term in exclusion_terms)def run_disabled_inactive_user_reclaim(users):    recommendations = []    for user in users:        if is_excluded_user(user):            continue        assigned_licenses = user.get("assignedLicenses", [])        if not assigned_licenses:            continue        disabled = user.get("accountEnabled") is False        last_login_days = user.get("lastLoginDaysAgo", 999)        eligible = disabled or last_login_days > 90        if not eligible:            continue        trust = score_user_reclaim_candidate(user)        for licence in assigned_licenses:            sku = licence.get("skuId", "UNKNOWN")            monthly_cost = DEFAULT_LICENSE_COSTS.get(sku, 25.00)            recommendations.append({                "user_email": user.get("userPrincipalName"),                "licence_sku": sku,                "monthly_cost": monthly_cost,                "reason": "Disabled or inactive user with assigned licence",                "trust_score": trust["trust_score"],                "execution_status": trust["execution_status"],                "recommended_action": "REMOVE_LICENCE",                "trust_breakdown": trust,                "raw_user": user,            })    return recommendations

services/outcome_ledger.py
from models.database import SessionLocal, OutcomeLedgerdef log_outcome(    user_email,    action,    licence_sku,    before_cost,    after_cost,    approved,    executed,    evidence,):    db = SessionLocal()    monthly_saving = before_cost - after_cost    outcome = OutcomeLedger(        user_email=user_email,        action=action,        licence_sku=licence_sku,        before_cost=before_cost,        after_cost=after_cost,        monthly_saving=monthly_saving,        approved=approved,        executed=executed,        evidence=evidence,    )    db.add(outcome)    db.commit()    db.refresh(outcome)    db.close()    return outcome

services/execution_service.py
from services.outcome_ledger import log_outcomedef approve_and_execute_recommendation(recommendation):    """    Phase 1 MVP:    - Manual approval only    - Simulated execution first    - Later replace with real Graph licence removal call    """    if recommendation.execution_status not in ["APPROVAL_REQUIRED"]:        raise ValueError("Recommendation is not approved for execution")    evidence = {        "execution_mode": "MANUAL_APPROVAL_REQUIRED",        "execution_type": "SIMULATED",        "before": {            "user_email": recommendation.user_email,            "licence_sku": recommendation.licence_sku,            "monthly_cost": recommendation.monthly_cost,        },        "after": {            "licence_removed": True,            "monthly_cost": 0,        },    }    outcome = log_outcome(        user_email=recommendation.user_email,        action=recommendation.recommended_action,        licence_sku=recommendation.licence_sku,        before_cost=recommendation.monthly_cost,        after_cost=0,        approved=True,        executed=True,        evidence=evidence,    )    return outcome
